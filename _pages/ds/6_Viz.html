            autopct='%1.1f%%', startangle=90)
axes[1].set_title('지역별 비율 (Pie)')

# 도넛 차트
wedge_props = dict(width=0.5)
axes[2].pie(origin_counts, labels=origin_counts.index,
            autopct='%1.1f%%', startangle=90, wedgeprops=wedge_props)
axes[2].set_title('지역별 비율 (Donut)')

plt.show()
